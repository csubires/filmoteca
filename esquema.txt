CREATE TABLE IF NOT EXISTS "genre" (
	"id_genre"	INTEGER PRIMARY KEY AUTOINCREMENT,
	"name"	TEXT NOT NULL UNIQUE,
	"pathfolder"	TEXT NOT NULL UNIQUE,
	"num_movies"	INTEGER DEFAULT 0,
	"local_size"	INTEGER,
	"local_size_str"	TEXT,
	"local_duration"	INTEGER,
	"local_duration_str"	TEXT,
	"is_subgenre"	INTEGER DEFAULT 0,
	"hash_folder"	TEXT,
	"report_date"	TEXT
);
CREATE TABLE sqlite_sequence(name,seq);
CREATE TABLE IF NOT EXISTS "country" (
	"id_country"	INTEGER PRIMARY KEY AUTOINCREMENT,
	"name"	TEXT NOT NULL UNIQUE,
	"code"	TEXT DEFAULT 'n/a',
	"flag"	TEXT
);
CREATE TABLE IF NOT EXISTS "rating" (
	"id_rating"	INTEGER PRIMARY KEY AUTOINCREMENT,
	"src_img"	TEXT,
	"url"	TEXT,
	"title"	TEXT,
	"year"	INTEGER,
	"rating"	REAL,
	"is_present"	INTEGER DEFAULT (0),
	CONSTRAINT "rating_UN" UNIQUE("title","year")
);
CREATE TABLE IF NOT EXISTS "report" (
	"id_report"	INTEGER PRIMARY KEY AUTOINCREMENT,
	"report_date"	TEXT NOT NULL,
	"num_movies"	INTEGER,
	"num_genres"	INTEGER,
	"global_size"	INTEGER,
	"global_size_str"	TEXT,
	"global_duration"	INTEGER,
	"global_duration_str"	TEXT,
	"file_extensions"	TEXT,
	"add_recent"	INTEGER,
	"manual_stop"	INTEGER DEFAULT 0,
	"num_files"	INTEGER,
	"hdd_code"	INTEGER DEFAULT 0,
	"real_size"	TEXT
);
CREATE TABLE IF NOT EXISTS "data" (
	url_end TEXT,
	date_end TEXT,
	"id_data" INTEGER NOT NULL
, npseries INTEGER DEFAULT (1));
CREATE TABLE movies (
	id_movie INTEGER PRIMARY KEY AUTOINCREMENT,
	title TEXT NOT NULL,
	realtitle TEXT,
	"year" INTEGER,
	quality TEXT,
	extension TEXT,
	"size" INTEGER,
	size_str TEXT,
	duration INTEGER,
	duration_str TEXT,
	pathfile TEXT,
	resolution TEXT,
	fps REAL,
	urldesc TEXT,
	ratings REAL DEFAULT (0.0),
	urlpicture TEXT,
	censure INTEGER DEFAULT (0),
	file_created TEXT,
	report_date TEXT,
	id_genre INTEGER NOT NULL,
	id_subgenre INTEGER DEFAULT (-1),
	id_country INTEGER,
	hdd_code INTEGER DEFAULT (99) NOT NULL, deleted INTEGER DEFAULT (0),
	CONSTRAINT FK_movies_country FOREIGN KEY (id_country) REFERENCES country(id_country),
	CONSTRAINT FK_movies_genre_2 FOREIGN KEY (id_genre) REFERENCES genre(id_genre),
	CONSTRAINT FK_movies_genre_3 FOREIGN KEY (id_subgenre) REFERENCES genre(id_genre),
	CONSTRAINT movies_movies_FK FOREIGN KEY (id_movie) REFERENCES movies(id_movie) ON DELETE SET NULL ON UPDATE CASCADE,
	CONSTRAINT movies_movies_FK_1 FOREIGN KEY (id_movie) REFERENCES movies(id_movie) ON DELETE SET NULL ON UPDATE SET NULL
);
CREATE INDEX idx_movies_title ON movies(title);
CREATE INDEX idx_movies_year ON movies(year);
CREATE INDEX idx_movies_id_genre ON movies(id_genre);
CREATE INDEX idx_movies_id_subgenre ON movies(id_subgenre);
CREATE INDEX idx_movies_id_country ON movies(id_country);
CREATE INDEX idx_movies_pathfile ON movies(pathfile);
CREATE INDEX idx_movies_report_date ON movies(report_date);
CREATE INDEX idx_movies_ratings ON movies(ratings);
CREATE INDEX idx_movies_hdd_code ON movies(hdd_code);
CREATE INDEX idx_movies_extension ON movies(extension);
CREATE INDEX idx_movies_resolution ON movies(resolution);
CREATE INDEX idx_movies_quality ON movies(quality);
CREATE INDEX idx_movies_genre_report ON movies(id_genre, report_date);
CREATE INDEX idx_movies_country_rating ON movies(id_country, ratings);
CREATE INDEX idx_movies_hdd_report ON movies(hdd_code, report_date);
CREATE INDEX idx_movies_year_rating ON movies(year, ratings);
CREATE INDEX idx_genre_name ON genre(name);
CREATE INDEX idx_genre_pathfolder ON genre(pathfolder);
CREATE INDEX idx_genre_is_subgenre ON genre(is_subgenre);
CREATE INDEX idx_genre_report_date ON genre(report_date);
CREATE INDEX idx_genre_hash_folder ON genre(hash_folder);
CREATE INDEX idx_country_name ON country(name);
CREATE INDEX idx_country_code ON country(code);
CREATE INDEX idx_rating_title_year ON rating(title, year);
CREATE INDEX idx_rating_rating_val ON rating(rating);
CREATE INDEX idx_rating_is_present ON rating(is_present);
CREATE INDEX idx_report_date ON report(report_date);
CREATE INDEX idx_report_hdd_code ON report(hdd_code);
CREATE INDEX idx_data_id_data ON data(id_data);
CREATE INDEX idx_data_date_end ON data(date_end);
CREATE TABLE IF NOT EXISTS "user" (
	"id_user"	INTEGER PRIMARY KEY AUTOINCREMENT,
	"name"	TEXT NOT NULL,
	"password"	TEXT NOT NULL,
	"ip"	TEXT,
	"agent"	TEXT,
	"email"	TEXT NOT NULL UNIQUE,
	"role"	TEXT DEFAULT 'user',
	"created_at"	TEXT NOT NULL DEFAULT CURRENT_TIMESTAMP,
	"last_login"	TEXT,
	CONSTRAINT "user_UN" UNIQUE("name","email")
);
CREATE UNIQUE INDEX idx_user_email ON user(email);
CREATE UNIQUE INDEX idx_user_name ON user(name);
CREATE TABLE IF NOT EXISTS "torrent_cache" (
	"id_torrent_cache" INTEGER PRIMARY KEY AUTOINCREMENT,
	"date_cached" TEXT NOT NULL UNIQUE,
	"movies_json" TEXT,
	"series_json" TEXT,
	"url_end" TEXT,
	"npseries" INTEGER DEFAULT (1),
	"created_at" TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
